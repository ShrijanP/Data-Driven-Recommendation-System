# Function for recommendation system

def colab_recommend(prod_name):
    # index fetch
    index = np.where(pt.index==prod_name)[0][0]
    similar_items = sorted(list(enumerate(sim_score[index])),reverse=True,key=lambda x:x[1])[1:6]
    
    data = []
    for i in similar_items:
        item = []
        temp_df = reviews[reviews['Product Name'] == pt.index[i[0]]]
        item.extend(list(temp_df.drop_duplicates('Product Name')['Product Name'].values))
        item.extend(list(temp_df.drop_duplicates('Product Name')['Retail Price'].values))
        item.extend(list(temp_df.drop_duplicates('Product Name')['Discount'].values))
        
        data.append(item)
    
    return data

from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import linear_kernel

tfidf = TfidfVectorizer(stop_words='english')
tfidf_matrix = tfidf.fit_transform(reviews['Product Name'])

cosine_sim = linear_kernel(tfidf_matrix, tfidf_matrix)

def content_recommend(product_name, cosine_sim=cosine_sim):
    idx = reviews[reviews['Product Name'] == product_name].index[0]
    sim_scores = list(enumerate(cosine_sim[idx]))
    sim_scores = sorted(sim_scores, key=lambda x: x[1], reverse=True)
    sim_scores = sim_scores[1:6]
    product_indices = [i[0] for i in sim_scores]
    return reviews['Product Name'].iloc[product_indices]
